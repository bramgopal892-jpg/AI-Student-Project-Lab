<script>
const canvas = document.getElementById('simulationCanvas');
const ctx = canvas.getContext('2d');
let animationInterval;

function detectLanguage(text) {
    const hindiPattern = /[\u0900-\u097F]/;
    return hindiPattern.test(text) ? "Hindi" : "English";
}

function processIdea() {
    const input = document.getElementById('userInput').value.trim();
    if(input === "") { alert("Kripya apna idea ya question likhe."); return; }

    const language = detectLanguage(input);
    const responseDiv = document.getElementById('response');
    let responseText = language === "Hindi" ?
        `AI Response (Hindi): Aapka idea '${input}' par kaam shuru kiya ja raha hai. Arduino code aur virtual video next step me dikhaya jayega.` :
        `AI Response (English): Your idea '${input}' is being processed. Arduino code and virtual video will be shown in the next step.`;
    responseDiv.innerHTML = `<b>${responseText}</b>`;

    showProject(input);
    startSimulation(input);
}

function showProject(input) {
    const projectDiv = document.getElementById('projectSection');
    let code = `int led = 13;\nvoid setup() {\n  pinMode(led, OUTPUT);\n}\nvoid loop() {\n  digitalWrite(led, HIGH);\n  delay(1000);\n  digitalWrite(led, LOW);\n  delay(1000);\n}`;
    let components = "Arduino UNO, LED, 220Œ© Resistor, Breadboard, Jumper Wires";

    if(input.toLowerCase().includes('motor')) {
        code += `\nint motor = 9;\npinMode(motor, OUTPUT);\nvoid loop() { digitalWrite(motor, HIGH); delay(1000); digitalWrite(motor, LOW); }`;
        components += ", Motor";
    }
    if(input.toLowerCase().includes('buzzer')) {
        code += `\nint buzzer = 8;\npinMode(buzzer, OUTPUT);\nvoid loop() { digitalWrite(buzzer, HIGH); delay(500); digitalWrite(buzzer, LOW); }`;
        components += ", Buzzer";
    }

    projectDiv.innerHTML = `<h3>Arduino Code & Components</h3><pre>${code}</pre><p><b>Components:</b> ${components}</p>`;
}

function startSimulation(input) {
    clearInterval(animationInterval);
    let ledState=false, motorState=false, buzzerState=false;
    animationInterval = setInterval(() => {
        ctx.clearRect(0,0,canvas.width,canvas.height);
        ctx.font = "14px Arial";

        // LED
        if(input.toLowerCase().includes('led')) {
            ledState = !ledState;
            ctx.fillStyle = ledState ? 'green' : 'red';
            ctx.beginPath(); ctx.arc(100,100,20,0,Math.PI*2); ctx.fill();
            ctx.fillStyle='black'; ctx.fillText('LED', 90,140);
        }

        // Motor
        if(input.toLowerCase().includes('motor')) {
            motorState = !motorState;
            ctx.save(); ctx.translate(250,100); ctx.rotate(motorState?0.3:-0.3);
            ctx.fillStyle='orange'; ctx.fillRect(-20,-20,40,40); ctx.restore();
            ctx.fillStyle='black'; ctx.fillText('Motor', 230,140);
        }

        // Buzzer
        if(input.toLowerCase().includes('buzzer')) {
            buzzerState = !buzzerState;
            ctx.fillStyle = buzzerState ? 'yellow' : 'blue'; ctx.fillRect(380,80,40,40);
            ctx.fillStyle='black'; ctx.fillText('Buzzer', 370,140);
        }

    }, 1000);
}
